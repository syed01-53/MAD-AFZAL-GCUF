package com.example.signup

import android.annotation.SuppressLint
import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.view.View
import android.widget.CheckBox
import android.widget.EditText
import android.widget.RadioButton
import android.widget.Toast

class MainActivity : AppCompatActivity() {
  private  lateinit var name : EditText
    private lateinit var email : EditText
    private lateinit var password : EditText
    private lateinit var output : EditText
    private lateinit var confirm : EditText
    private lateinit var city : EditText
    private  lateinit var country : EditText
    private  lateinit var date : EditText
    private  lateinit var cricket : CheckBox
    private  lateinit var hockey : CheckBox
    private  lateinit var male : RadioButton
    private  lateinit var female : RadioButton
    private  lateinit var beginner : RadioButton
    private  lateinit var skillFull : RadioButton

    @SuppressLint("MissingInflatedId")
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        name=findViewById(R.id.name)
        email=findViewById(R.id.email)
        password=findViewById(R.id.password)
        confirm=findViewById(R.id.confirm)
        city=findViewById(R.id.city)
        country=findViewById(R.id.country)
          date=findViewById(R.id.dob)
        male=findViewById(R.id.male)
        female =findViewById(R.id.female)
        beginner=findViewById(R.id.beginner)
        skillFull=findViewById(R.id.professional)
        cricket=findViewById(R.id.checkbox_cheese)
        hockey=findViewById(R.id.checkbox_meat)
         output =findViewById(R.id.result)

    }

// clear function
    fun clear(view: View) {
        clearEditText(name)
        clearEditText(email)
        clearEditText(password)
        clearEditText(confirm)
        clearEditText(city)
        clearEditText(country)
        clearEditText(date)
      clearCheckBox(cricket)
        clearCheckBox(hockey)
        clearRadioBtn(male)
        clearRadioBtn(female)
        clearRadioBtn(beginner)
        clearRadioBtn(skillFull)
    }

    private fun clearEditText(editText: EditText) {
        editText.text?.clear()
    }
    private fun clearCheckBox(checkBox: CheckBox){
        checkBox.isChecked = false
    }
    private fun clearRadioBtn(radioButton: RadioButton){
        radioButton.isChecked=false
    }

    fun Submitt(view: View) {
        //evey field never be empty
       if (!name.text.isNullOrBlank() && !password.text.isNullOrBlank() && !email.text.isNullOrBlank()
           && !confirm.text.isNullOrBlank() && !date.text.isNullOrBlank() && !city.text.isNullOrBlank()
           && !country.text.isNullOrBlank()
           ){
           if(cricket.isChecked == true || hockey.isChecked ==true)
           {
             if (male.isChecked== true || female.isChecked ==true)  {
                 if(beginner.isChecked ==true || skillFull.isChecked ==true )
                 {
                     if (password.text.toString()== confirm.text.toString() ){
                         Toast.makeText(this,"Success sign" , Toast.LENGTH_SHORT).show()

                      print()

                     }
                     else{
                         Toast.makeText(this,"please correct password" , Toast.LENGTH_SHORT).show()
                     }

                 }
                 else{
                     Toast.makeText(this,"skill level missing" , Toast.LENGTH_SHORT).show()
                 }
             }
               else {
                 Toast.makeText(this,"male 0r female" , Toast.LENGTH_SHORT).show()
               }
           }
           else{
               Toast.makeText(this,"sport is missing" , Toast.LENGTH_SHORT).show()
           }

       }
        else{
           Toast.makeText(this,"field empty" , Toast.LENGTH_SHORT).show()
       }
    }



    private fun print(){
      var result = "$name"
                output.setText(result)
    }



}
